#!/usr/bin/env python3
"""
LIVE WEIGHT TRAINER
Continuously updates neural weights from new Q&A facts

Monitors: training_data/user_comm_qa.txt
Updates: 498D EM field weights + 82D emotion weights
Runs: Background daemon, saves weights every 10 new facts
"""

import sys
import json
import time
import numpy as np
from pathlib import Path
from datetime import datetime

BASE = Path(__file__).parent.parent
sys.path.append(str(BASE))

# Import models
from models.minimal_llm_498d import MinimalLLM498D
from tokenizer.text_encoder import ConstraintLatticeEncoder

class LiveWeightTrainer:
    def __init__(self):
        print("🔥 Initializing Live Weight Trainer...")
        print()
        
        # Paths
        self.qa_file = BASE / "training_data" / "user_comm_qa.txt"
        self.em_weights_file = BASE / "models" / "minimal_llm_498d_weights_em field.npz"
        
        # Load 498D model
        self.model = MinimalLLM498D()
        self._load_weights()
        
        # Encoder
        self.encoder = ConstraintLatticeEncoder()
        self.vectors = np.load(BASE / "tokenizer" / "token_vectors_498d.npy")
        
        with open(BASE / "tokenizer" / "word_token_map.json", 'r') as f:
            data = json.load(f)
            self.word_to_tokens = data['word_to_tokens']
        
        # Training params
        self.learning_rate = 0.01
        self.batch_size = 10  # Update weights every 10 new facts
        
        # Track what we've trained on
        self.last_trained_line = self._get_last_trained_line()
        
        print(f"  ✅ Model loaded")
        print(f"  ✅ Last trained on line: {self.last_trained_line}")
        print(f"  ✅ Learning rate: {self.learning_rate}")
        print()
    
    def _load_weights(self):
        """Load existing weights"""
        if self.em_weights_file.exists():
            data = np.load(self.em_weights_file)
            self.model.W1 = data['W1']
            self.model.W2 = data['W2']
            self.model.b1 = data['b1']
            self.model.b2 = data['b2']
        else:
            print("  ⚠️  No existing weights, starting fresh")
    
    def _save_weights(self):
        """Save updated weights"""
        np.savez(
            self.em_weights_file,
            W1=self.model.W1,
            W2=self.model.W2,
            b1=self.model.b1,
            b2=self.model.b2
        )
        print(f"  💾 Weights saved: {datetime.now().strftime('%H:%M:%S')}")
    
    def _get_last_trained_line(self):
        """Get the last line we trained on"""
        tracker_file = BASE / "memory" / "training_tracker.json"
        
        if tracker_file.exists():
            with open(tracker_file, 'r') as f:
                data = json.load(f)
                return data.get('last_line', 0)
        
        return 0
    
    def _update_training_tracker(self, line_num):
        """Update what line we've trained up to"""
        tracker_file = BASE / "memory" / "training_tracker.json"
        tracker_file.parent.mkdir(parents=True, exist_ok=True)
        
        with open(tracker_file, 'w') as f:
            json.dump({
                'last_line': line_num,
                'timestamp': datetime.utcnow().isoformat() + 'Z'
            }, f, indent=4)
    
    def text_to_vector(self, text):
        """Convert text to 498D vector (average of word vectors)"""
        words = text.lower().split()
        vectors = []
        
        for word in words:
            if word in self.word_to_tokens:
                token_ids = self.word_to_tokens[word]
                for tid in token_ids[:1]:  # Use first token
                    if tid < len(self.vectors):
                        vectors.append(self.vectors[tid])
        
        if not vectors:
            # Return zero vector if no words found
            return np.zeros(498)
        
        return np.mean(vectors, axis=0)
    
    def get_new_training_pairs(self):
        """Get Q&A pairs we haven't trained on yet"""
        if not self.qa_file.exists():
            return []
        
        new_pairs = []
        
        with open(self.qa_file, 'r') as f:
            lines = f.readlines()
        
        for i, line in enumerate(lines[self.last_trained_line:], start=self.last_trained_line):
            line = line.strip()
            if not line or line.startswith('#'):
                continue
            
            if '|' in line:
                q, a = line.split('|', 1)
                new_pairs.append((q.strip(), a.strip(), i + 1))
        
        return new_pairs
    
    def train_on_pair(self, question, answer):
        """Train model on one Q&A pair"""
        
        # Encode question and answer as vectors
        q_vec = self.text_to_vector(question)
        a_vec = self.text_to_vector(answer)
        
        # Forward pass
        output = self.model.forward(q_vec)
        
        # Calculate loss (MSE between output and answer vector)
        loss = np.mean((output - a_vec) ** 2)
        
        # Backward pass (simplified gradient descent)
        output_error = output - a_vec
        
        # Update W2 and b2
        hidden = self.model.h1  # From forward pass
        self.model.W2 -= self.learning_rate * np.outer(hidden, output_error)
        self.model.b2 -= self.learning_rate * output_error
        
        # Update W1 and b1 (backprop through hidden layer)
        hidden_error = np.dot(self.model.W2, output_error)
        hidden_error = hidden_error * (hidden > 0)  # ReLU derivative
        
        self.model.W1 -= self.learning_rate * np.outer(q_vec, hidden_error)
        self.model.b1 -= self.learning_rate * hidden_error
        
        return loss
    
    def run_training_cycle(self):
        """One training cycle - check for new facts and train"""
        
        new_pairs = self.get_new_training_pairs()
        
        if not new_pairs:
            return 0
        
        print(f"\n🎓 Training on {len(new_pairs)} new facts...")
        
        total_loss = 0
        
        for q, a, line_num in new_pairs:
            loss = self.train_on_pair(q, a)
            total_loss += loss
            
            # Update tracker
            self.last_trained_line = line_num
        
        avg_loss = total_loss / len(new_pairs)
        
        print(f"  📊 Average loss: {avg_loss:.4f}")
        print(f"  ✅ Trained up to line: {self.last_trained_line}")
        
        # Save weights
        self._save_weights()
        self._update_training_tracker(self.last_trained_line)
        
        return len(new_pairs)
    
    def run_daemon(self, check_interval=10):
        """Run as background daemon, checking every N seconds"""
        
        print("=" * 70)
        print("🔥 LIVE WEIGHT TRAINER DAEMON")
        print("=" * 70)
        print()
        print(f"Monitoring: {self.qa_file}")
        print(f"Check interval: {check_interval} seconds")
        print()
        print("Press Ctrl+C to stop")
        print()
        
        try:
            while True:
                trained = self.run_training_cycle()
                
                if trained > 0:
                    print(f"  ⚡ Model improved! (+{trained} facts)\n")
                
                time.sleep(check_interval)
                
        except KeyboardInterrupt:
            print("\n\n🛑 Stopping trainer...")
            self._save_weights()
            print("✅ Final weights saved")
            print("Goodbye. 👋\n")

"""
DNA-Tokenizer: dna_token.py
Core token generator — RGB to binary + AM/FM to binary
Proof of concept — simple, clean, testable
DNA base pairs: A=AM_freq  T=Token_color  C=Color_hue  G=FM_freq
"""

import struct
import hashlib
import json
from datetime import datetime

# Simple frequency ranges aligned with 498D hue frequency concept
# Hue maps 0-360 to 400-700 THz (visible light) — we mirror that logic
AM_MIN = 530.0    # kHz
AM_MAX = 1700.0   # kHz
FM_MIN = 87.5     # MHz
FM_MAX = 108.0    # MHz

def rgb_to_binary(r, g, b):
    """Convert RGB values to binary strings — T base pair"""
    r_bin = format(r, '08b')
    g_bin = format(g, '08b')
    b_bin = format(b, '08b')
    return r_bin, g_bin, b_bin

def freq_to_binary(freq, freq_min, freq_max, bits=16):
    """
    Map a frequency value to binary — A and G base pairs
    Normalizes freq to 0-65535 range then encodes to bits
    """
    normalized = (freq - freq_min) / (freq_max - freq_min)
    normalized = max(0.0, min(1.0, normalized))
    int_val = int(normalized * ((2 ** bits) - 1))
    return format(int_val, f'0{bits}b'), int_val

def hue_from_rgb(r, g, b):
    """
    Derive hue from RGB — C base pair
    Simple hue calculation 0-360
    """
    r_norm = r / 255.0
    g_norm = g / 255.0
    b_norm = b / 255.0
    cmax = max(r_norm, g_norm, b_norm)
    cmin = min(r_norm, g_norm, b_norm)
    delta = cmax - cmin
    if delta == 0:
        hue = 0.0
    elif cmax == r_norm:
        hue = 60 * (((g_norm - b_norm) / delta) % 6)
    elif cmax == g_norm:
        hue = 60 * (((b_norm - r_norm) / delta) + 2)
    else:
        hue = 60 * (((r_norm - g_norm) / delta) + 4)
    return round(hue, 2)

def hue_to_freq(hue):
    """Map hue 0-360 to visible light frequency 400-700 THz — mirrors 498D"""
    return round(400.0 + (hue / 360.0) * 300.0, 4)

def generate_dna_token(r, g, b, am_freq, fm_freq):
    """
    Generate a complete DNA token from RGB + AM/FM frequencies
    
    Structure:
        A = AM frequency binary  (16 bits)
        T = Token RGB binary     (24 bits — 8R + 8G + 8B)
        C = Color hue + hue_freq (derived from RGB)
        G = FM frequency binary  (16 bits)
    
    Returns full token dict with binary strand and hash address
    """
    # T base pair — RGB binary
    r_bin, g_bin, b_bin = rgb_to_binary(r, g, b)
    t_strand = r_bin + g_bin + b_bin  # 24 bits

    # A base pair — AM frequency binary
    a_bin, am_int = freq_to_binary(am_freq, AM_MIN, AM_MAX)

    # G base pair — FM frequency binary
    g_bin_freq, fm_int = freq_to_binary(fm_freq, FM_MIN, FM_MAX)

    # C base pair — hue derived from RGB + visible light frequency
    hue = hue_from_rgb(r, g, b)
    hue_freq = hue_to_freq(hue)
    c_strand = format(int(hue), '09b')  # 9 bits covers 0-360

    # Full DNA strand — ATCG order
    full_strand = a_bin + t_strand + c_strand + g_bin_freq

    # Hash address — SHA-256 of the full strand (the token's unique address)
    hash_address = hashlib.sha256(full_strand.encode()).hexdigest()

    token = {
        "dna_strand": {
            "A": a_bin,           # AM frequency binary
            "T": t_strand,        # RGB binary
            "C": c_strand,        # Hue binary
            "G": g_bin_freq       # FM frequency binary
        },
        "values": {
            "rgb": [r, g, b],
            "am_freq_khz": am_freq,
            "fm_freq_mhz": fm_freq,
            "hue": hue,
            "hue_freq_thz": hue_freq
        },
        "full_strand": full_strand,
        "hash_address": hash_address,
        "timestamp": datetime.utcnow().isoformat() + "Z",
        "strand_length_bits": len(full_strand)
    }

    return token

def save_token(token, filename=None):
    """Save token to output/tokens/ directory"""
    if not filename:
        ts = datetime.utcnow().strftime("%Y%m%d_%H%M%S")
        filename = f"token_{ts}.json"
    path = f"output/tokens/{filename}"
    with open(path, 'w') as f:
        json.dump(token, f, indent=2)
    return path
